package main

import "fmt"

func main() {
	process()
	fmt.Println("Returned from process")
}

func process() {
	defer func() {
		// if r := recover(); r != nil { // if there is no nil then some panic happened
		r := recover()
		if r != nil {
			fmt.Println("Recovered:", r)
		}
	}()
	fmt.Println("Start Process")
	panic("Something went wrong")
	defer fmt.Println("This shall not run")
}
