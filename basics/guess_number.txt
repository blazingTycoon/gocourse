package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {

	source := rand.NewSource(time.Now().UnixNano())

	random := rand.New(source)

	// Generate a random number between 1 and 100

	target := random.Intn(100) + 1

	fmt.Println("Welcome to the Guessing Game")
	fmt.Println("I have guessed a number between 1 and 100")
	fmt.Println("Try to guess it")

	var guess int
	for {
		fmt.Println("Enter the guess : ")
		fmt.Scanln(&guess)

		if target == guess {
			fmt.Println("Congratulations you have guess it right !")
			break
		} else if target < guess {
			fmt.Println("You need to try a lower value")
		} else {
			fmt.Println("You need to try a larger value")
		}
	}
}
