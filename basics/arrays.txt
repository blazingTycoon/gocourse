package main

import "fmt"

func main() {
	// var arrayName [size]elementType
	var numbers [5]int
	fmt.Println(numbers)

	numbers[4] = 20
	fmt.Println(numbers)

	numbers[0] = 9
	fmt.Println(numbers)

	fruits := [4]string{"apple", "banana", "orange", "grapes"}
	fmt.Println("Fruits array :", fruits)
	fmt.Println(fruits[2])

	originalArray := [3]int{1, 2, 3}
	copiedArray := originalArray //copying happens by copy-by-value so effects occur if the copied array is changed
	copiedArray[0] = 100

	fmt.Println("Original Array: ", originalArray, "Copied Array: ", copiedArray)
	//Printing all the elements in the array
	for idx, value := range numbers {
		fmt.Printf("Numbers[%d]: %d\n", idx, value)
	}
	// _ -> blank identifier
	a, _ := someFunction()
	fmt.Println(a)
	// fmt.Println(b)

	b := 2
	_ = b // assigning b to an underscore if we are not  using the variable that is declared.

	fmt.Println("The length of the array is : ", len(numbers))

	//Comparing arrays

	array1 := [3]int{1, 2, 3}
	array2 := [3]int{1, 1, 3}
	if array1 == array2 {
		fmt.Println("Both the arrays are equal")
	} else {
		fmt.Println("Both the arrays are not equal")
	}

	// Multi-dimensional arrays

	var matrix [3][3]int = [3][3]int{
		{1, 2, 3},
		{4, 5, 6},
		{7, 8, 9},
	}
	// Printing the 2 d array
	for i := range matrix {
		for j := range matrix {
			fmt.Printf("%d ", matrix[i][j])
		}
		fmt.Println()
	}
	//Python type printing
	fmt.Println(matrix)

	originalArray = [3]int{1, 2, 3}
	var copiedNewArray *[3]int // -> pointer to an address of array that has three elements
	copiedNewArray = &originalArray

	copiedNewArray[0] = 100
	fmt.Println("Original Array: ", originalArray, "Copied Array: ", *copiedNewArray)

}

func someFunction() (int, int) {
	return 1, 2
}
