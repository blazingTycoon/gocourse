package main

import "fmt"

func main() {
	a := factorial(5)
	fmt.Println(a)
	fmt.Println(sumDigits(12))
}

func factorial(n int) int {
	// Base case
	if n == 0 {
		return 1
	}
	// Recurisve Case
	return n * factorial(n-1)
}

func sumDigits(n int) int {
	// Base Case
	if n < 10 {
		return n
	}
	// Recursive Case
	return n%10 + sumDigits(n/10)
}
